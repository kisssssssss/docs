## 原理

> https://www.hello-algo.com/chapter_sorting/insertion_sort/

![](https://pic1.zhimg.com/v2-be81c151f38d8923fe1ede31ac530ac4_b.webp)

时间复杂度为 $O(n^2)$

## C++实现

```cpp
void InsertionSort(int num[], int length) {
  if (length <= 1) {
    return;
  } else {
    for (int i = 1; i < length; i++) {
      // 插入的值
      int insertVal = num[i];
      // 插入的位置
      int insertIndex = 0;
      // 找到要插入的位置
      for (int j = 0; j < i; j++) {
        if (insertVal < num[j]) {
          // 保存要插入的位置
          insertIndex = j;
          // 将数组往后移一个位置
          for (int temp = i; temp > j; temp--) {
            num[temp] = num[temp - 1];
          }
          // 插入
          num[insertIndex] = insertVal;
          break;
        }
      }
    }
  }
}
```

## TypeScript实现

```ts
function InsertionSort(num: number[]): number[] {
  if (num.length <= 1) {
    return num;
  } else {
    let result: number[] = [num[0]];
    // 遍历要插入的元素
    for (let i = 1; i < num.length; i++) {
      // 遍历已排序数组
      for (let j = 0; j < result.length; j++) {
        // 找到插入位置
        if (num[i] < result[j]) {
          if (j === 0) {
            result.unshift(num[i]);
          } else {
            result.splice(j, 0, num[i]);
          }
          break;
        } else if (j === result.length - 1) {
          result.push(num[i]);
          break; 
        }
      }
    }
    return result;
  }
}
```
